import { Playground } from 'docz'
import { calendarActions, useCalendarDispatch, Calendar, CalendarProvider} from '@use-date-input/core';
import { adapter as dateAdapter } from "@use-date-input/date-fns-adapter";
import { DemoButton } from "../../gatsby-theme-docz/components/DemoButton";
import { DemoButtonBar } from "../../gatsby-theme-docz/components/DemoButtonBar";

# useCalendarDispatch
`useCalendarDispatch` returns the configured reducers dispatcher, along with a helper API for common actions. 

```javascript
const {
    dispatch,
    mouseClickOutside,
    navigateNext,
    navigatePrevious,
    selectDate,
    setAnimating,
    setEnableKeyboardNavigation,
    setKeyboardCursor,
    setMouseCursor,
    setEndDate,
    setStartDate,
    setOpen,
    setKeyPress,
    setVisibleFromDate
} = useCalendarDispatch();
```

## Using the dispatcher
Using the dispatcher we can dispatch any action to the reducers.

<Playground>
{() => {
    const MyPanel = () => {
        const { dispatch } = useCalendarDispatch();
        const handleClick = () => dispatch({ type: calendarActions.navigateNext });
        return (
            <DemoButtonBar>
                <DemoButton onClick={handleClick}>NEXT MONTH</DemoButton>
            </DemoButtonBar>
        );
    };
    return (
        <CalendarProvider adapter={dateAdapter}> 
            <MyPanel/>
            <Calendar/>
        </CalendarProvider>
    );
}}
</Playground>

## Using the helper API
Using the dispatcher we can dispatch the most common actions, through helper methods.

<Playground>

            </DemoButtonBar>
